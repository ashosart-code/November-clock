// Define the exact end of November 2025 (in your time zone, assuming EST) // Note: JavaScript Date object months are 0-indexed (Nov is month 10) const END_DATE = new Date('November 30, 2025 23:59:59 EST'); // Scaling Factor (SF24): 1 simulated minute = 30 real minutes const SCALING_FACTOR = 30; function updateClock() { const now = new Date(); // 1. Calculate Real Time Remaining (in milliseconds) const timeRemaining_ms = END_DATE.getTime() - now.getTime(); // Check if the countdown is over if (timeRemaining_ms <= 0) { document.getElementById('simulatedTime').innerText = "00:00:00 - DONE"; clearInterval(timerInterval); // Stop the clock return; } // 2. Convert to Simulated Time // A. Convert real remaining milliseconds to real remaining minutes (divide by 60,000) const realMinutesRemaining = timeRemaining_ms / 60000; // B. Scale the remaining minutes using the 24-hour factor (30 real minutes per 1 simulated minute) const simulatedMinutesRemaining = realMinutesRemaining / SCALING_FACTOR; // 3. Deconstruct Simulated Time into H:M:S // Total simulated seconds remaining const totalSimulatedSeconds = Math.floor(simulatedMinutesRemaining * 60); // Calculate hours, minutes, and seconds for the display const hours = Math.floor(totalSimulatedSeconds / 3600); const minutes = Math.floor((totalSimulatedSeconds % 3600) / 60); const seconds = totalSimulatedSeconds % 60; // 4. Format the Output const h = String(hours).padStart(2, '0'); const m = String(minutes).padStart(2, '0'); const s = String(seconds).padStart(2, '0'); // Display the simulated time (time remaining in the 24-hour simulation) document.getElementById('simulatedTime').innerText = `${h}:${m}:${s}`; } // Start the clock and update it every 1000 milliseconds (1 second) const timerInterval = setInterval(updateClock, 1000); updateClock(); // Run immediately to avoid a 1-second delay
